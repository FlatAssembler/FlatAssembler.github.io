  14 // Declare the "Queen" structure and write relevant functions.
  15 Structure Queen Consists Of
  16   Integer32 row, column;
  17 EndStructure
  18 
  19 Function areQueensInTheSameColumn(QueenPointer first, QueenPointer second)
  20   Which Returns Integer32 Does
  21     Return first->column = second->column;
  22 EndFunction
  23 
  24 Function areQueensInTheSameRow(QueenPointer first, QueenPointer second)
  25   Which Returns Integer32 Does
  26     Return first->row = second->row;
  27 EndFunction
  28 
  29 Function areQueensOnTheSameDiagonal(QueenPointer first,
  30                                     QueenPointer second)
  31   Which Returns Integer32 Does
  32     Return first->row + first->column = second->row + second->column or
  33            first->row - first->column = second->row - second->column;
  34 EndFunction
  35 
  36 Function areQueensAttackingEachOther(QueenPointer first,
  37                                      QueenPointer second)
  38   Which Returns Integer32 Does
  39     Return areQueensInTheSameRow(first, second) or
  40            areQueensInTheSameColumn(first, second) or
  41            areQueensOnTheSameDiagonal(first, second);
  42 EndFunction

  64 // Let's write the function that JavaScript is supposed to call...
  65 Function nQueensPuzzle(Integer32 n) Which Returns Integer32 Does
  66   clearScreen();
  67   If n < 1 or n > 12 Then
  68     printString("Please enter a number between 1 and 12!");
  69     Return -1;
  70   EndIf
  71   InstantiateStructure ChessBoard chessBoard;
  72   Character stringToBePrinted[64] := {0};
  73   CharacterPointer stringToBePrinted := AddressOf(stringToBePrinted[0]);
  74   strcat(stringToBePrinted, "Solving the n-queens puzzle for ");
  75   convertIntegerToString(stringToBePrinted + strlen(stringToBePrinted),
  76                          n);
  77   strcat(stringToBePrinted,":\n");
  78   printString(stringToBePrinted);
  79   Integer32 result := recursiveFunction(AddressOf(chessBoard), n);
  80   stringToBePrinted[0] := 0;
  81   strcat(stringToBePrinted, "Found ");
  82   convertIntegerToString(stringToBePrinted + strlen(stringToBePrinted),
  83                          result);
  84   strcat(stringToBePrinted, " solutions!");
  85   printString(stringToBePrinted);
  86   Return result;
  87 EndFunction

  89 // I guess moving this code out of "recursiveFunction" makes the
  90 // code more legible.
  91 Function printAsASolution(ChessBoardPointer chessBoard)
  92   Which Returns Nothing Does
  93     Character stringToBePrinted[32] := {0};
  94     Integer32 i := 0;
  95     While i < chessBoard->length Loop
  96       Character stringToBeAdded[8];
  97       stringToBeAdded[0] := 'A' + chessBoard->queens[i].column;
  98       convertIntegerToString(AddressOf(stringToBeAdded[1]),
  99                              chessBoard->queens[i].row + 1);
 100       strcat(AddressOf(stringToBeAdded[0]), " ");
 101       strcat(AddressOf(stringToBePrinted[0]),
 102              AddressOf(stringToBeAdded[0]));
 103       i += 1;
 104     EndWhile
 105     strcat(AddressOf(stringToBePrinted[0]), "\n");
 106     printString(AddressOf(stringToBePrinted[0]));
 107 EndFunction

 109 // Now, let's implement the brute-force algorithm.
 110 Function recursiveFunction(ChessBoardPointer chessBoard,
 111                            Integer32 n) Which Returns Integer32 Does
 112   // First, do some sanity checks useful for debugging...
 113   If chessBoard->length > n Then
 114     printString("Bug: Chessboard length too large!");
 115     Return 0;
 116   EndIf
 117   Integer32 i := 0, j := 0;
 118   While i < chessBoard->length Loop
 119     If chessBoard->queens[i].column < 0 or
 120        chessBoard->queens[i].row    < 0 or
 121        chessBoard->queens[i].column > n or
 122        chessBoard->queens[i].row    > n Then
 123       printString("Bug: Corrupt chessboard!");
 124       Return 0;
 125     EndIf
 126     i += 1;
 127   EndWhile
 128   // Check if there is a contradiction (queens attacking
 129   // each other) in what we have thus far...
 130   i := j := 0;
 131   While i < chessBoard->length Loop
 132     j := i + 1;
 133     While j < chessBoard->length Loop
 134       If not(i = j) and areQueensAttackingEachOther(
 135                           AddressOf(chessBoard->queens[i]),
 136                           AddressOf(chessBoard->queens[j])
 137                         ) Then
 138         Return 0;
 139       EndIf
 140       j += 1;
 141     EndWhile
 142     i += 1;
 143   EndWhile
 144   // Check if this is a solution...
 145   If chessBoard->length = n Then
 146     printAsASolution(chessBoard);
 147     Return 1;
 148   EndIf
 149   // If this is not a complete solution, but there are no contradictions
 150   // in it, branch the recursion into searching for complete solutions
 151   // based on this one.
 152   Integer32 result := 0;
 153   i := 0;
 154   While i<n Loop
 155     InstantiateStructure ChessBoard newChessBoard := ValueAt(chessBoard);
 156     newChessBoard.length += 1;
 157     newChessBoard.queens[chessBoard->length].column := chessBoard->length;
 158     newChessBoard.queens[chessBoard->length].row := i;
 159     result += recursiveFunction(AddressOf(newChessBoard), n);
 160     i += 1;
 161   EndWhile
 162   Return result;
 163 EndFunction

 165 // Now go the helper functions related to string manipulation,
 166 // copied from the Dragon Curve program. They are named the same
 167 // as the corresponding functions in the standard C library.
 168 Function strlen(CharacterPointer str) Which Returns Integer32 Does
 169      Integer32 length := 0;
 170      While ValueAt(str + length) Loop
 171          length := length + 1;
 172      EndWhile
 173      Return length;
 174 EndFunction
 175 
 176 Function strcpy(CharacterPointer dest,
 177                 CharacterPointer src) Which Returns Nothing Does
 178     While ValueAt(src) Loop
 179         ValueAt(dest) := ValueAt(src);
 180         dest          :=     dest + 1;
 181         src           :=      src + 1;
 182      EndWhile
 183     ValueAt(dest) := 0;
 184 EndFunction
 185 
 186 Function strcat(CharacterPointer dest,
 187                 CharacterPointer src) Which Returns Nothing Does
 188     strcpy(dest + strlen(dest), src);
 189 EndFunction

 191 Function reverseString(CharacterPointer string) Which Returns Nothing Does
 192     CharacterPointer pointerToLastCharacter := string + strlen(string) - 1;
 193     While pointerToLastCharacter - string > 0 Loop
 194         Character tmp                   := ValueAt(string);
 195         ValueAt(string)                 := ValueAt(pointerToLastCharacter);
 196         ValueAt(pointerToLastCharacter) := tmp;
 197         string                          := string + 1;
 198         pointerToLastCharacter          := pointerToLastCharacter - 1;
 199     EndWhile
 200 EndFunction

 202 Function convertIntegerToString(CharacterPointer string,
 203                                 Integer32 number)
 204     Which Returns Nothing Does
 205     Integer32 isNumberNegative := 0;
 206     If number < 0 Then
 207         number           := -number;
 208         isNumberNegative :=       1;
 209     EndIf
 210     Integer32 i := 0;
 211     While number > 9 Loop
 212         ValueAt(string + i) := '0' + mod(number, 10);
 213         number              :=           number / 10;
 214         i                   :=                 i + 1;
 215     EndWhile
 216     ValueAt(string + i) := '0' + number;
 217     i                   :=        i + 1;
 218     If isNumberNegative Then
 219         ValueAt(string + i) :=   '-';
 220         i                   := i + 1;
 221     EndIf
 222     ValueAt(string + i) := 0;
 223     reverseString(string);
 224 EndFunction
