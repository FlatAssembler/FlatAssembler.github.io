  30 Function newTreeNode() Which Returns PointerToTreeNode Does {
  31   Integer16 i : = 0;
  32   While i < 32 Loop {
  33     If not(isTreeNodeUsed[i]) Then {
  34       treeNodes[i].character : = 0;
  35       treeNodes[i].leftChild : = treeNodes[i].rightChild
  36           : = PointerToTreeNode(0);
  37       treeNodes[i].code[0] : = 0;
  38       treeNodes[i].frequencyOfCharacter : = 0;
  39       isTreeNodeUsed[i] : = 1;
  40       If NDEBUG = 0 Then {
  41         Character stringToBePrinted[64] : = {0};
  42         strcat(AddressOf(stringToBePrinted[0]),
  43                "NDEBUG: Allocating the TreeNode #");
  44         convertIntegerToString(AddressOf(stringToBePrinted[0]) +
  45                                    strlen(AddressOf(stringToBePrinted[0])),
  46                                i);
  47         strcat(AddressOf(stringToBePrinted[0]), "\n");
  48         printString(AddressOf(stringToBePrinted[0]));
  49       }
  50       EndIf;
  51       Return AddressOf(treeNodes[i]);
  52     }
  53     EndIf;
  54     i += 1;
  55   }
  56   EndWhile;
  57   noMoreFreeMemory();
  58 }
  59 EndFunction;

  62   If not(AddressOf(treeNodes[0]) <= treeNode <= AddressOf(treeNodes[32 - 1]))
  63       Then {
  64     segmentationFault();
  65   }
  66   EndIf;
  67   If NDEBUG = 0 Then {
  68     Character stringToBePrinted[64] : = {0};

  70     convertIntegerToString(AddressOf(stringToBePrinted[0]) +
  71                                strlen(AddressOf(stringToBePrinted[0])),
  72                            (treeNode - AddressOf(treeNodes[0])) /
  73                                SizeOf(TreeNode));
  74     strcat(AddressOf(stringToBePrinted[0]), "\n");
  75     printString(AddressOf(stringToBePrinted[0]));
  76   }
  77   EndIf;
  78   isTreeNodeUsed[(treeNode - AddressOf(treeNodes[0])) / SizeOf(TreeNode)] : = 0;
  79 }

 108 Function reverseString(PointerToCharacter string) Which Returns Nothing Does {
 109   PointerToCharacter pointerToLastCharacter : = string + strlen(string) - 1;
 110   While pointerToLastCharacter - string > 0 Loop {
 111     Character tmp : = ValueAt(string);
 112     ValueAt(string) : = ValueAt(pointerToLastCharacter);
 113     ValueAt(pointerToLastCharacter) : = tmp;
 114   string:
 115     = string + 1;
 116   pointerToLastCharacter:
 117     = pointerToLastCharacter - 1;
 118   }
 119   EndWhile
 120 }
 121 EndFunction;

 123 Function convertIntegerToString(PointerToCharacter string,
 124                                 Integer32 number) Which Returns Nothing Does {
 125   Integer32 isNumberNegative : = 0;
 126   If number < 0 Then {
 127   number:
 128     = -number;
 129   isNumberNegative:
 130     = 1;
 131   }
 132   EndIf Integer32 i : = 0;
 133   While number >= 10 Loop {
 134     ValueAt(string + i) : = '0' + mod(number, 10);
 135     number /= 10;
 136     i += 1;
 137   }
 138   EndWhile;
 139   ValueAt(string + i) : = '0' + number;
 140   i += 1;
 141   If isNumberNegative Then {
 142     ValueAt(string + i) : = '-';
 143     i += 1;
 144   }
 145   EndIf ValueAt(string + i) : = 0;
 146   reverseString(string);
 147 }

 152 Function assignCode(PointerToCharacter currentCode,
 153                     PointerToTreeNode treeNode) Which Returns Nothing Does {
 154   If NDEBUG = 0 Then {
 155     Character stringToBePrinted[64] : = {0};
 156     strcat(AddressOf(stringToBePrinted[0]), "NDEBUG: Assigning the code \"");

 159     convertIntegerToString(AddressOf(stringToBePrinted[0]) +
 160                                strlen(AddressOf(stringToBePrinted[0])),
 161                            (treeNode - AddressOf(treeNodes[0])) /
 162                                SizeOf(TreeNode));
 163     strcat(AddressOf(stringToBePrinted[0]), ".\n");
 164     printString(AddressOf(stringToBePrinted[0]));
 165 
 166     stringToBePrinted[0] : = 0;
 167     strcat(AddressOf(stringToBePrinted[0]),
 168            "NDEBUG: Its left child is the TreeNode #");
 169     If treeNode->leftChild Then {
 170       convertIntegerToString(AddressOf(stringToBePrinted[0]) +
 171                                  strlen(AddressOf(stringToBePrinted[0])),
 172                              (treeNode->leftChild - AddressOf(treeNodes[0])) /
 173                                  SizeOf(TreeNode));
 174     }
 175     Else { strcat(AddressOf(stringToBePrinted[0]), "null"); }
 176     EndIf;
 177     strcat(AddressOf(stringToBePrinted[0]), ".\n");
 178     printString(AddressOf(stringToBePrinted[0]));
 179 
 180     stringToBePrinted[0] : = 0;
 181     strcat(AddressOf(stringToBePrinted[0]),
 182            "NDEBUG: Its right child is the TreeNode #");
 183     If treeNode->leftChild Then {
 184       convertIntegerToString(AddressOf(stringToBePrinted[0]) +
 185                                  strlen(AddressOf(stringToBePrinted[0])),
 186                              (treeNode->rightChild - AddressOf(treeNodes[0])) /
 187                                  SizeOf(TreeNode));
 188     }
 189     Else { strcat(AddressOf(stringToBePrinted[0]), "null"); }
 190     EndIf;
 191     strcat(AddressOf(stringToBePrinted[0]), ".\n");
 192     printString(AddressOf(stringToBePrinted[0]));
 193   }
 194   EndIf;
 195 
 196   If not(AddressOf(treeNodes[0]) <= treeNode <= AddressOf(treeNodes[32 - 1]))
 197       Then {
 198     segmentationFault();
 199   }
 200   EndIf;
 201 
 202   strcpy(AddressOf(treeNode->code[0]), currentCode);
 203 
 204   Character leftCode[16] : = {0}, rightCode[16] : = {0};
 205 
 206   strcpy(AddressOf(leftCode[0]), currentCode);
 207   strcat(AddressOf(leftCode[0]), "0");
 208   strcpy(AddressOf(rightCode[0]), currentCode);
 209   strcat(AddressOf(rightCode[0]), "1");
 210

 212     assignCode(AddressOf(leftCode[0]), treeNode->leftChild);
 213   }
 214   EndIf;
 215   If treeNode->rightChild Then {
 216     assignCode(AddressOf(rightCode[0]), treeNode->rightChild);
 217   }
 218   EndIf;
 219 
 220   If treeNode->character Then {
 221     mapOfCodes[treeNode->character] : = AddressOf(treeNode->code[0]);
 222     Character codeToPrint[32] : = {0};
 223     codeToPrint[0] : = '\'';
 224     codeToPrint[1] : = treeNode->character;
 225     codeToPrint[2] : = '\'';
 226     codeToPrint[3] : = '=';
 227     codeToPrint[4] : = '>';
 228     codeToPrint[5] : = 0;
 229     strcat(AddressOf(codeToPrint[0]), AddressOf(treeNode->code[0]));
 230     strcat(AddressOf(codeToPrint[0]), "\n");
 231     printString(AddressOf(codeToPrint[0]));
 232   }
 233   EndIf;
 234 }

 237 Function freeUpTheTree(PointerToTreeNode tree) Which Returns Nothing Does {
 238   If tree->leftChild Then { freeTreeNode(tree->leftChild); }
 239   EndIf;
 240   If tree->rightChild Then { freeTreeNode(tree->rightChild); }
 241   EndIf;
 242   freeTreeNode(tree);
 243 }
