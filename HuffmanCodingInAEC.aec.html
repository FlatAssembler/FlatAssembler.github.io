  35 Function isTreeNodeWithinBounds(
  36     PointerToTreeNode treeNode) Which Returns Integer32 Does {
  37   Return AddressOf(treeNodes[0]) <= treeNode <= AddressOf(treeNodes[64 - 1]) and
  38       mod(treeNode - AddressOf(treeNodes[0]), SizeOf(TreeNode)) =
  39       0; // In C, testing whether the mod is equal to 0 would not be necessary,
  40          // but, since AEC supports unaligned access, it is necessary here.

  51 Function newTreeNode() Which Returns PointerToTreeNode Does {
  52   Integer16 i : = 0;
  53   While i < 64 Loop {
  54     If not(isTreeNodeUsed[i]) Then {
  55       treeNodes[i].character : = 0;
  56       treeNodes[i].leftChild : = treeNodes[i].rightChild
  57           : = PointerToTreeNode(0);
  58       treeNodes[i].code[0] : = 0;
  59       treeNodes[i].frequencyOfCharacter : = 0;
  60       isTreeNodeUsed[i] : = 1;
  61       If NDEBUG = 0 Then {
  62         Character stringToBePrinted[64] : = {0};
  63         strcat(AddressOf(stringToBePrinted[0]),
  64                "NDEBUG: Allocating the TreeNode #");
  65         convertIntegerToString(AddressOf(stringToBePrinted[0]) +
  66                                    strlen(AddressOf(stringToBePrinted[0])),
  67                                i);
  68         strcat(AddressOf(stringToBePrinted[0]), "\n");
  69         printString(AddressOf(stringToBePrinted[0]));
  70       }
  71       EndIf;
  72       Return AddressOf(treeNodes[i]);
  73     }
  74     EndIf;
  75     i += 1;
  76   }
  77   EndWhile;
  78   noMoreFreeMemory();
  79 }
  80 EndFunction;

  91                                SizeOf(TreeNode));

  93     printString(AddressOf(stringToBePrinted[0]));
  94   }
  95   EndIf;

  97 }

 176 Function convertIntegerToString(PointerToCharacter string,
 177                                 Integer32 number) Which Returns Nothing Does {
 178   Integer32 isNumberNegative : = 0;
 179   If number < 0 Then {
 180   number:
 181     = -number;
 182   isNumberNegative:
 183     = 1;
 184   }
 185   EndIf;
 186   Integer32 i : = 0;
 187   While number >= 10 Loop {
 188     ValueAt(string + i) : = '0' + mod(number, 10);
 189     number /= 10;
 190     i += 1;
 191   }
 192   EndWhile;
 193   ValueAt(string + i) : = '0' + number;
 194   i += 1;
 195   If isNumberNegative Then {
 196     ValueAt(string + i) : = '-';
 197     i += 1;
 198   }
 199   EndIf;
 200   ValueAt(string + i) : = 0;
 201   reverseString(string);
 202 }
 203 EndFunction;

 205 Function convertDecimalToString(PointerToCharacter string,
 206                                 Decimal32 number) Which Returns Nothing Does {
 207   convertIntegerToString(string, number);
 208   number -= Integer32(number);
 209   If number < 0 Then { number *= -1; }
 210   EndIf;
 211   If number = 0 Then { Return; }
 212   EndIf;
 213   strcat(string, ".");
 214   Integer16 numberOfDecimals : = 0;
 215   While numberOfDecimals<3 and number> 0 Loop {
 216     numberOfDecimals += 1;
 217     number *= 10;
 218     Character stringWithTheNewDigit[2];
 219     stringWithTheNewDigit[0] : = stringWithTheNewDigit[1] : = 0;
 220     stringWithTheNewDigit[0] : = '0' + number;
 221     strcat(string, AddressOf(stringWithTheNewDigit[0]));
 222     number -= Integer32(number);
 223   }
 224   EndWhile;
 225 }
 226 EndFunction;

 324 Decimal32 PRECISION : = 500;
 325 Function ln(Decimal32 x) Which Returns Decimal32 Does {
 326   // Natural logarithm is the integral of 1/x from 1 to x, highschool math.
 327   Decimal32 sum : = 0, epsilon : = (x - 1) / (5 * PRECISION), i : = 1;
 328   While(epsilon > 0 and i < x) or (epsilon < 0 and i > x) Loop {
 329     sum += epsilon / i;
 330     i += epsilon;
 331   }
 332   EndWhile;
 333   Return sum;
 334 }
 335 EndFunction;
