  29 Function printFloat32(Decimal32 x) Which Returns Nothing Is Declared;
  30 /*
  31  * We cannot make it external because we are not actually using WASI, and the
  32  * compiler thinks we are, and will output code that will not run in NodeJS.
  33  * Therefore, we need to import the function from JavaScript manually in
  34  * inline assembly. Not hard, fortunately.
  35  */
  36 asm("(func $printFloat32 (import \"JavaScript\" \"printFloat32\") (param f32))");

  38 // Should return 0.
  39 Function namedArgumentsTest() Which Returns Integer32 Does
  40   // Since 'a' is set to default to 1 in the declaration of the
  41   // "solveQuadraticEquation" function, it does not need to be written
  42   // if it is indeed 1 when calling that function.
  43   InstantiateStructure QuadraticEquationSolution solution[1 * 2 * 3];
  44   solveQuadraticEquation(a := 1 , b := -1, c := -1, AddressOf(solution[0]));
  45   solveQuadraticEquation(c := -1, b := -1, solution := AddressOf(solution[1]));
  46   solveQuadraticEquation(b := -1, c := -1, solution := AddressOf(solution[2]));
  47   solveQuadraticEquation(b := -1, c := -1, a := 1 , AddressOf(solution[3]));
  48   solveQuadraticEquation(c := -1, a := 1 , b := -1, AddressOf(solution[4]));
  49   solveQuadraticEquation(c := -1, b := -1, solution := AddressOf(solution[5]));
  50   Integer32 iterator := 0;
  51   While iterator < 1*2*3 - 1 Loop
  52     If not(areStructuresEqual(AddressOf(solution[iterator]),
  53                               AddressOf(solution[iterator + 1]),
  54                               SizeOf(QuadraticEquationSolution))) Then
  55       Return iterator + 1;
  56     EndIf
  57     iterator += 1;
  58   EndWhile
  59   iterator := 0;
  60   While iterator < 1*2*3 - 1 Loop /*
  61                                    * This loop should have no effect,
  62                                    * but the compiler must not crash
  63                                    * when compiling it.
  64                                    */
  65     solution[iterator] := solution[iterator + 1];
  66     iterator += 1;
  67   EndWhile
  68   If not(solution[0].firstSolution.imaginary = 0 and
  69          solution[0].secondSolution.imaginary = 0 and
  70          abs(solution[0].firstSolution.real + goldenRatio - 1) < epsilon and
  71          abs(solution[0].secondSolution.real - goldenRatio) < epsilon) Then
  72     printFloat32(solution[0].firstSolution.real);
  73     printFloat32(solution[0].firstSolution.imaginary);
  74     printFloat32(solution[0].secondSolution.real);
  75     printFloat32(solution[0].secondSolution.imaginary);
  76     Return 7;
  77   EndIf
  78   Return 0;
  79 EndFunction

  82   Return x < 0 ? -x : x;
  83 EndFunction

  85 Function areStructuresEqual(CharacterPointer first, CharacterPointer second,
  86                             Integer32 size) Which Returns Integer32 Does
  87   Integer32 i := 0;
  88   While i < size Loop
  89     If not(ValueAt(first + i) = ValueAt(second + i)) Then
  90       Return 0;
  91     EndIf
  92     i += 1;
  93   EndWhile
  94   Return 1;
  95 EndFunction

 100 Function solveQuadraticEquation(Decimal32 a, Decimal32 b, Decimal32 c,
 101                                 QuadraticEquationSolutionPointer solution)
 102                                 Which Returns Nothing Does
 103   solution->firstSolution.real := -b;
 104   solution->firstSolution.imaginary := 0;
 105   solution->secondSolution := solution->firstSolution;
 106   InstantiateStructure ComplexNumber rootOfDiscriminant;
 107   sqrt(b*b - 4*a*c, AddressOf(rootOfDiscriminant));
 108   solution->firstSolution.real -= rootOfDiscriminant.real;
 109   solution->firstSolution.imaginary -= rootOfDiscriminant.imaginary;
 110   solution->secondSolution.real += rootOfDiscriminant.real;
 111   solution->secondSolution.imaginary += rootOfDiscriminant.imaginary;
 112   solution->firstSolution.real /= 2*a;
 113   solution->firstSolution.imaginary /= 2*a;
 114   solution->secondSolution.real /= 2*a;
 115   solution->secondSolution.imaginary /= 2*a;
 116 EndFunction

 118 Function sqrt(Decimal32 x, ComplexNumberPointer result)
 119         Which Returns Nothing Does
 120   If x < 0 Then
 121     result->real := 0;
 122     result->imaginary := asm_f32("(f32.sqrt\n"
 123                                  "\t(f32.sub\n"
 124                                  "\t\t(f32.const 0)\n"
 125                                  "\t\t(local.get 0)\n"
 126                                  "\t)\n"
 127                                  ")");
 128   Else
 129     result->real := asm_f32("(f32.sqrt\n"
 130                             "\t(local.get 0)\n"
 131                             ")");
 132     result->imaginary := 0;
 133   EndIf
 134 EndFunction
