1 /*
   2  * Ovo je moj pokušaj implementacije Routh-Hurwitzovog kriterija stabilnosti
   3  * linearnih vremenski nepromjenjivih sustava u programskom jeziku AEC.
   4  * U biti, to je algoritam da se pomoću matrica provjeri koliko korijena neki
   5  * polinom ima desno od imaginarne osi (dakle, s pozitivnim realnim dijelom),
   6  * bez potrebe da se oni računaju (što je u principu nemoguće za polinome
   7  * višeg reda).
   8  */
   9 
  10  #target WASI // Nećemo pozivati nikakve JavaScriptine funkcije, pa će
  11               // potprogram pisan u JavaScriptu biti jednostavniji ako AEC
  12               // compiler misli da ciljamo WASI, a ne browser.
  13 
  14 Integer16 stupanj_polinoma;
  15 Decimal64 polinom[20], matrica[20 * 20];
  16 
  17 Function postavi_stupanj_polinoma(Integer16 n) Which Returns Nothing Does
  18   stupanj_polinoma := n;
  19 EndFunction
  20 
  21 Function postavi_polinom(Integer16 indeks, Decimal64 broj) Which Returns Nothing
  22   Does
  23     polinom[indeks] := broj;
  24 EndFunction
  25 
  26 Function signum(Decimal64 x) Which Returns Integer16 Does
  27   Return x < 0 ? -1 :
  28          x = 0 ? 0 :
  29          1;
  30 EndFunction
  31 
  32 Function provjeri_nuzne_uvjete() Which Returns Integer16 Does
  33   Integer16 i := 1;
  34   While i < stupanj_polinoma + 1 Loop
  35     If not(signum(polinom[i - 1]) = signum(polinom[i])) Then
  36       Return 0;
  37     EndIf
  38     i += 1;
  39   EndWhile
  40   Return 1;
  41 EndFunction
  42 
  43 Function f(Integer16 i, Integer16 j) Which Returns Integer16 Does
  44   // Za pretvaranje indeksa dvodimenzionalnog polja u indeks jednodimenzionalnog
  45   // polja. Kada u svoj AEC compiler još nisam implementirao dvodimenzionalna
  46   // polja...
  47   Return 20 * i + j;
  48 EndFunction
  49 
  50 Function popuni_matricu() Which Returns Integer16 Does
  51   Integer16 broj_stupaca :=
  52               (stupanj_polinoma + 1) / 2 + mod(stupanj_polinoma + 1, 2),
  53             broj_redaka := stupanj_polinoma + 1;
  54   Integer16 i := 0;
  55   //Popunimo matricu prvo not-a-numbersima...
  56   While i < broj_redaka Loop
  57     Integer16 j := 0;
  58     While j < broj_stupaca Loop
  59       matrica[f(i, j)] := 0. / 0.;
  60       j += 1;
  61     EndWhile
  62     i += 1;
  63   EndWhile
  64   //Zatim idemo primjeniti Hurwitzov algoritam...
  65   i := 0;
  66   While i < broj_redaka Loop
  67     Integer16 j := 0;
  68     While j < broj_stupaca Loop
  69       If i = 0 Then // Prvi redak
  70         matrica[f(i, j)] := polinom[j * 2];
  71       ElseIf i = 1 Then // Drugi redak
  72         matrica[f(i, j)] := (j * 2 + 1 < stupanj_polinoma + 1) ?
  73                             polinom[j * 2 + 1] :
  74                             0;
  75       Else // Ostali reci...
  76         If matrica[f(i - 1, 0)] = 0 Then
  77           //TODO: Implementirati što se radi u posebnim slučajevima...
  78           Return 0;
  79         EndIf
  80         matrica[f(i, j)] := (matrica[f(i - 1, 0)] *
  81                             (j + 1 < broj_stupaca ?
  82                              matrica[f(i - 2, j + 1)] : 0) -
  83                             (matrica[f(i - 2, 0)] *
  84                             (j + 1 < broj_stupaca ?
  85                              matrica[f(i - 1, j + 1)] : 0))) /
  86                             matrica[f(i - 1 , 0)];
  87       EndIf
  88       j += 1;
  89     EndWhile
  90     i += 1;
  91   EndWhile
  92   If matrica[f(broj_redaka - 1, 0)] = polinom[stupanj_polinoma] Then
  93     Return 1;
  94   EndIf
  95   Return 0;
  96 EndFunction
  97 
  98 Function broj_korijena_u_desnoj_poluravnini() Which Returns Integer16 Does
  99   Integer16 i := 1, brojac := 0;
 100   While i < stupanj_polinoma + 1 Loop
 101     brojac += not(signum(matrica[f(i, 0)]) = signum(matrica[f(i - 1, 0)]));
 102     i += 1;
 103   EndWhile
 104   Return brojac;
 105 EndFunction
 106 
 107 Function broj_iz_matrice(Integer16 i, Integer16 j) Which Returns Decimal64 Does
 108   //Ova se funkcija poziva iz JavaScripta kad se crta tablica...
 109   Return matrica[f(i, j)];
 110 EndFunction
