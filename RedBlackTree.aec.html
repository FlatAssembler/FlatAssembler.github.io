  42 Function reverseString(PointerToCharacter string) Which Returns Nothing Does
  43   PointerToCharacter pointerToLastCharacter := string + strlen(string) - 1;
  44   While pointerToLastCharacter - string > 0 Loop
  45     Character tmp := ValueAt(string);
  46     ValueAt(string) := ValueAt(pointerToLastCharacter);
  47     ValueAt(pointerToLastCharacter) := tmp;
  48     string += 1;
  49     pointerToLastCharacter -= 1;
  50   EndWhile

  53 Function convertIntegerToString(PointerToCharacter string, Integer32 number) 
  54     Which Returns Nothing Does
  55   Integer32 isNumberNegative := 0;
  56   If number < 0 Then
  57     number := -number;
  58     isNumberNegative := 1;
  59   EndIf
  60   Integer32 i := 0;
  61   While number >= 10 Loop
  62     ValueAt(string + i) := '0' + mod(number, 10);
  63     number /= 10;
  64     i += 1;
  65   EndWhile
  66   ValueAt(string + i) := '0' + number;
  67   i += 1;
  68   If isNumberNegative Then 
  69     ValueAt(string + i) := '-';
  70     i += 1;
  71   EndIf
  72   ValueAt(string + i) := 0;
  73   reverseString(string);
  74 EndFunction

  91 Function isRedBlackTreeNodeWithinBounds(
  92          PointerToRedBlackTreeNode node) Which Returns Integer32 Does
  93   Return AddressOf(redBlackTreeNodes[0]) <= node <= AddressOf(redBlackTreeNodes[128 - 1]) and
  94       mod(node - AddressOf(redBlackTreeNodes[0]), SizeOf(RedBlackTreeNode)) = 0;

  97 Function newRedBlackTreeNode() Which Returns PointerToRedBlackTreeNode Does
  98   Integer16 i := 0;
  99   While i < 128 Loop
 100     If not(isRedBlackTreeNodeUsed[i]) Then
 101       redBlackTreeNodes[i].key := 0;
 102       redBlackTreeNodes[i].leftChild :=
 103          redBlackTreeNodes[i].rightChild :=
 104            PointerToRedBlackTreeNode(0);
 105       redBlackTreeNodes[i].color := 'R';
 106       redBlackTreeNodes[i].x := redBlackTreeNodes[i].y := 0;
 107       isRedBlackTreeNodeUsed[i] := 1;
 108       Return AddressOf(redBlackTreeNodes[i]);
 109     EndIf
 110     i += 1;
 111   EndWhile
 112   noMoreFreeMemory();
 113 EndFunction

 186 Function Insert(PointerToRedBlackTreeNode node, Integer32 X)
 187     Which Returns PointerToRedBlackTreeNode Does
 188   Integer32 flag := 0;
 189     If not(node) Then
 190       PointerToRedBlackTreeNode ret := newRedBlackTreeNode();
 191       ret->key := X;
 192       Return ret;
 193     EndIf
 194     If X < node->key Then
 195         node->leftChild := Insert(node->leftChild, X);
 196         node->leftChild->parent := node;
 197         If not(node=root) Then
 198           If node->color = 'R' and node->leftChild->color = 'R' Then
 199             flag := 1;
 200           EndIf
 201         EndIf
 202     ElseIf X > node->key Then
 203         node->rightChild := Insert(node->rightChild, X);
 204         node->rightChild->parent := node;
 205         If not(node=root) Then
 206            If node->color = 'R' and node->rightChild->color = 'R' Then
 207              flag := 1;
 208            EndIf
 209         EndIf
 210     Else
 211       Return node;
 212     EndIf
 213  
 214     If ll Then
 215         node := LRotate( node );
 216         node->color := 'B';
 217         node->leftChild->color := 'R';
 218         ll := 0;
 219     ElseIf rr Then
 220         node := RRotate( node );
 221         node->color := 'B';
 222         node->rightChild->color := 'R';
 223         rr  := 0;
 224     ElseIf rl Then
 225         node->rightChild := RRotate( node->rightChild );
 226         node->rightChild->parent := node;
 227         node := LRotate( node );
 228         node->color := 'B';
 229         node->leftChild->color := 'R';
 230         rl := 0;
 231     ElseIf lr Then
 232         node->leftChild := LRotate( node->leftChild );
 233         node->leftChild->parent := node;
 234         node := RRotate( node );
 235         node->color := 'B';
 236         node->rightChild->color := 'R';
 237         lr := 0;
 238     EndIf
 239 
 240     If flag Then  // RED-RED conflict
 241         If node->parent->rightChild = node Then
 242             If node->parent->leftChild = PointerToRedBlackTreeNode(0) or node->parent->leftChild->color = 'B' Then
 243                 If not(node->leftChild = PointerToRedBlackTreeNode(0)) and node->leftChild->color = 'R' Then
 244                   rl := 1;
 245                 ElseIf not(node->rightChild = PointerToRedBlackTreeNode(0)) and node->rightChild->color = 'R' Then
 246                   ll := 1;
 247                 EndIf
 248             Else
 249                 node->parent->leftChild->color := 'B';
 250                 node->color := 'B';
 251                 If not(node->parent = root) Then
 252                   node->parent->color := 'R';
 253                 EndIf
 254             EndIf
 255         Else
 256             If node->parent->rightChild = PointerToRedBlackTreeNode(0) or node->parent->rightChild->color = 'B' Then
 257                 If not(node->leftChild = PointerToRedBlackTreeNode(0)) and node->leftChild->color = 'R' Then
 258                   rr := 1;
 259                 ElseIf not(node->rightChild = PointerToRedBlackTreeNode(0)) and node->rightChild->color = 'R' Then
 260                   lr := 1;
 261                 EndIf
 262             Else
 263                 node->parent->rightChild->color := 'B';
 264                 node->color := 'B';
 265                 If not(node->parent = root) Then
 266                   node->parent->color := 'R';
 267                 EndIf
 268             EndIf
 269         EndIf
 270         flag := 0;
 271     EndIf
 272     Return node; 
 273 EndFunction

 344 Function FixDoubleBlack(PointerToRedBlackTreeNode x) Which Returns Nothing Does
 345    If x = root Then
 346      Return;
 347    EndIf
 348    PointerToRedBlackTreeNode sibling := Sibling( x ), parent := x->parent;
 349    If not(sibling) Then
 350      FixDoubleBlack(parent);
 351    Else
 352      If sibling->color = 'R' Then
 353        parent->color := 'R';
 354        sibling->color := 'B';
 355        If sibling = sibling->parent->leftChild Then
 356          rightRotate(parent);
 357        Else 
 358            leftRotate(parent);
 359        EndIf
 360        FixDoubleBlack(x); 
 361      Else
 362        If not(sibling->leftChild = PointerToRedBlackTreeNode(0)) and sibling->leftChild->color = 'R' Then
 363          If sibling = sibling->parent->leftChild Then
 364            sibling->leftChild->color := sibling->color;
 365            sibling->color := parent->color;
 366            rightRotate(parent); 
 367          Else
 368            sibling->leftChild->color := parent->color;
 369            rightRotate(sibling);
 370            leftRotate(parent);
 371          EndIf
 372          parent->color := 'B'; 
 373        ElseIf not(sibling->rightChild = PointerToRedBlackTreeNode(0)) and sibling->rightChild->color='R' Then
 374          If sibling = sibling->parent->leftChild Then
 375            sibling->rightChild->color := parent->color;
 376            leftRotate(sibling);
 377            rightRotate(parent); 
 378          Else
 379            sibling->rightChild->color := sibling->color;
 380            sibling->color := parent->color;
 381            leftRotate(parent);
 382          EndIf
 383          parent->color := 'B';
 384        Else
 385          sibling->color := 'R';
 386          If parent->color = 'B' Then
 387            FixDoubleBlack(parent);
 388          Else
 389            parent->color := 'B';
 390          EndIf
 391        EndIf
 392     EndIf
 393   EndIf
 394 EndFunction

 405 Function ReplaceRBT(PointerToRedBlackTreeNode t)
 406     Which Returns PointerToRedBlackTreeNode Does
 407   If not(t->leftChild = PointerToRedBlackTreeNode(0)) and
 408      not(t->rightChild = PointerToRedBlackTreeNode(0)) Then
 409     Return SmallestNode(t->rightChild);
 410   ElseIf not(t->leftChild) and not(t->rightChild) Then
 411     Return 0;
 412   ElseIf t->leftChild Then
 413     Return t->leftChild;
 414   Else
 415     Return t->rightChild;
 416   EndIf
 417   asm("unreachable");
 418 EndFunction

 420 Function DeleteNode(PointerToRedBlackTreeNode v)
 421     Which Returns Nothing Does
 422   PointerToRedBlackTreeNode u := ReplaceRBT(v);
 423   Integer32 uvBothBlack := (not(u) or u->color = 'B') and v->color = 'B';
 424   PointerToRedBlackTreeNode parent := v->parent;
 425   If not(u) Then
 426     If v = root Then
 427       root := PointerToRedBlackTreeNode(0);
 428     Else
 429       If uvBothBlack Then
 430         FixDoubleBlack(v);
 431       Else
 432         PointerToRedBlackTreeNode vsib := Sibling(v);
 433         If vsib Then
 434           vsib->color := 'R';
 435         EndIf
 436       EndIf
 437       If v = v->parent->leftChild Then
 438         parent->leftChild := PointerToRedBlackTreeNode(0);
 439       Else
 440         parent->rightChild := PointerToRedBlackTreeNode(0);
 441       EndIf
 442     EndIf
 443     freeRedBlackTreeNode(v);
 444     Return;
 445    EndIf
 446    If v->leftChild = PointerToRedBlackTreeNode(0)
 447       or v->rightChild = PointerToRedBlackTreeNode(0) Then
 448      If v = root Then
 449        v->key := u->key;
 450        v->leftChild := v->rightChild := PointerToRedBlackTreeNode(0);
 451        freeRedBlackTreeNode(u);
 452      Else
 453        If v = v->parent->leftChild Then
 454          parent->leftChild := u;
 455        Else
 456          parent->rightChild := u;
 457        EndIf
 458        freeRedBlackTreeNode(v);
 459        u->parent := parent;
 460        If uvBothBlack Then
 461          FixDoubleBlack(u);
 462        Else
 463          u->color := 'B';
 464        EndIf
 465      EndIf
 466      Return;
 467     EndIf
 468     Integer32 tmp := u->key;
 469     u->key := v->key;
 470     v->key := tmp;
 471     DeleteNode(u);
 472 EndFunction

 474 Function SearchRBT(
 475     PointerToRedBlackTreeNode node,
 476     Integer32 X,
 477     PointerToPointerToRedBlackTreeNode parent)
 478     Which Returns Integer32 Does
 479   If not(node) Then
 480     Return 0;
 481   ElseIf node->key = X Then
 482     Return 1;
 483   EndIf
 484   ValueAt(parent) := node;
 485   If node->key > X Then
 486     Return SearchRBT(node->leftChild, X, parent);
 487   Else
 488     Return SearchRBT(node->rightChild, X, parent);
 489   EndIf
 490   asm("unreachable");
 491 EndFunction

 493 Function DeleteRBT(PointerToRedBlackTreeNode node, Integer32 X) Which Returns Nothing Does
 494   If not(node) Then
 495     Return;
 496   EndIf
 497   PointerToRedBlackTreeNode parent := PointerToRedBlackTreeNode(0);
 498   If SearchRBT(node, X, AddressOf(parent)) Then
 499     If not(parent) Then
 500       DeleteNode(node);
 501     ElseIf not(parent->leftChild = PointerToRedBlackTreeNode(0)) and
 502            parent->leftChild->key = X Then
 503       DeleteNode(parent->leftChild);
 504     Else
 505       DeleteNode(parent->rightChild);
 506     EndIf
 507   Else
 508     printString("Node to be deleted has not been found.\n");
 509   EndIf
 510 EndFunction

 627 Function render() Which Returns Nothing Does
 628   clearScreen();
 629   root := PointerToRedBlackTreeNode(0);
 630   Integer32 i := 0;
 631   While i < numberOfKeys Loop
 632     If insert_or_delete[i] = 'I' Then
 633       Character stringToBePrinted[32] := {0};
 634       strcat(AddressOf(stringToBePrinted[0]), "Inserting ");
 635       appendIntegerToString(
 636         AddressOf(stringToBePrinted[0]),
 637         keys[i]);
 638       strcat(AddressOf(stringToBePrinted[0]), "...\n");
 639       printString(AddressOf(stringToBePrinted[0]));
 640 
 641       root := InsertRBT(root, keys[i]);
 642     ElseIf insert_or_delete[i] = 'D' Then
 643       Character stringToBePrinted[32] := {0};
 644       strcat(AddressOf(stringToBePrinted[0]), "Deleting ");
 645       appendIntegerToString(
 646         AddressOf(stringToBePrinted[0]),
 647         keys[i]);
 648       strcat(AddressOf(stringToBePrinted[0]), "...\n");
 649       printString(AddressOf(stringToBePrinted[0]));
 650 
 651       DeleteRBT(root, keys[i]);
 652     Else
 653       printString(R"RawString(The character array "insert_or_delete" can only contain characters 'I' and 'D'!
 654 )RawString");
 655       asm("unreachable");
 656     EndIf
 657     i += 1;
 658   EndWhile
 659 
 660   inorderCounter := 1;
 661   assignCoordinatesInorder(root, 0, 110, 90);
 662 
 663   InstantiateStructure Maxima globalMax;
 664   globalMax.minimumX := globalMax.maximumX := 0;
 665   If root Then
 666     globalMax.minimumX := root->x;
 667     globalMax.maximumX := root->x;
 668     findMinMaxX(root, AddressOf(globalMax));
 669   EndIf
 670 
 671   Integer32 margin := 20;
 672   Integer32 diagramWidth := (globalMax.maximumX - globalMax.minimumX) + 120 + margin * 2;
 673   Integer32 diagramHeight := (nodeHeight(root) + 1) * 90 + margin;
 674 
 675   setDiagramWidth(diagramWidth);
 676   setDiagramHeight(diagramHeight);
 677 
 678   drawRedBlackTree(root, globalMax.minimumX - margin);
 679 
 680   freeRedBlackTree(root);
 681   printString("Red Black Tree rendered.\n");
 682 EndFunction
