  12 Integer16 koliko_ljudi_ima_rodendan_na_taj_dan[1_000];
  13 // Recimo da nikome neće trebati izračun za "godinu" s više od tisuću dana.
  14 
  15 Function numericki_izracun_rodendanskog_paradoksa(
  16 			Integer16 koliko_godina_ima_dana       ,
  17 			Integer16 koliko_ima_ljudi             ,
  18 			Integer16 iznad_koliko_kolizija_brojimo,
  19 		) Which Returns Decimal32 Does
  20 	Integer32 koliko_smo_puta_izvrtili_simulaciju  := 100_000,
  21 		koliko_smo_puta_dobili_toliko_kolizija := 0      ,
  22 		vanjski_brojac                         := 0      ;
  23 	While vanjski_brojac < koliko_smo_puta_izvrtili_simulaciju Loop
  24 		Integer16 unutarnji_brojac := 0,
  25                 	dani_za_debuggiranje[1_000];
  26 		While unutarnji_brojac < koliko_godina_ima_dana Loop
  27 			koliko_ljudi_ima_rodendan_na_taj_dan[
  28 					unutarnji_brojac
  29 				]        := 0;
  30 			unutarnji_brojac += 1;
  31 		EndWhile
  32 		unutarnji_brojac := 0;
  33 		While unutarnji_brojac < koliko_ima_ljudi Loop
  34 			Integer16 nasumican_dan := mod(
  35 							nasumican_broj(),
  36 							koliko_godina_ima_dana
  37 						);
  38 			koliko_ljudi_ima_rodendan_na_taj_dan[
  39 							nasumican_dan
  40 					] += 1;
  41 			dani_za_debuggiranje[
  42 					unutarnji_brojac
  43 				]         := nasumican_dan;
  44 			unutarnji_brojac  += 1;
  45 		EndWhile
  46 		Integer16 jesmo_li_nasli_potreban_broj_kolizija := 0;
  47 		unutarnji_brojac := 0;
  48 		While unutarnji_brojac < koliko_godina_ima_dana Loop
  49 			If
  50 				koliko_ljudi_ima_rodendan_na_taj_dan[
  51 					unutarnji_brojac
  52 				] > iznad_koliko_kolizija_brojimo or
  53 				koliko_ljudi_ima_rodendan_na_taj_dan[
  54 					unutarnji_brojac
  55 				] = iznad_koliko_kolizija_brojimo
  56 			Then
  57 				jesmo_li_nasli_potreban_broj_kolizija := 1;	
  58 			EndIf
  59 			unutarnji_brojac += 1;
  60 		EndWhile
  61 		If jesmo_li_nasli_potreban_broj_kolizija Then
  62 			koliko_smo_puta_dobili_toliko_kolizija += 1;
  63 		ElseIf koliko_ima_ljudi > koliko_godina_ima_dana Then
  64 			printString("---");
  65 			unutarnji_brojac := 0;
  66 			While unutarnji_brojac < koliko_godina_ima_dana Loop
  67 				printInt(dani_za_debuggiranje[
  68 						unutarnji_brojac
  69 					]);
  70 				unutarnji_brojac += 1;
  71 			EndWhile
  72 			printString("---");
  73 		EndIf
  74 		vanjski_brojac += 1;
  75 	EndWhile
  76 	Return Decimal32(
  77 			koliko_smo_puta_dobili_toliko_kolizija
  78 		) / koliko_smo_puta_izvrtili_simulaciju;
  79 EndFunction

  81 Integer32 nasumican_broj := 9_907;
  82 Function nasumican_broj() Which Returns Integer16 Does
  83 	Return nasumican_broj := mod(nasumican_broj * 48_271, 2_147_483_647);
  84 	// http://cplusplus.com/reference/random/minstd_rand/
  85 EndFunction
