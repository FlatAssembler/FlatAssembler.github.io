  17 Function strlen(PointerToCharacter str) Which Returns Integer32 Does
  18   If str = 0 Then
  19     Return 0;
  20   EndIf
  21   Integer32 length := 0;
  22   While ValueAt(str + length) Loop
  23     length += 1;
  24   EndWhile
  25   Return length;
  26 EndFunction

  28 Function strcpy(PointerToCharacter dest, PointerToCharacter src) Which Returns Nothing Does
  29   While ValueAt(src) Loop
  30     ValueAt(dest) := ValueAt(src);
  31     dest += 1;
  32     src += 1;
  33   EndWhile
  34   ValueAt(dest) := 0;
  35 EndFunction

  41 Function reverseString(PointerToCharacter string) Which Returns Nothing Does
  42   PointerToCharacter pointerToLastCharacter := string + strlen(string) - 1;
  43   While pointerToLastCharacter - string > 0 Loop
  44     Character tmp := ValueAt(string);
  45     ValueAt(string) := ValueAt(pointerToLastCharacter);
  46     ValueAt(pointerToLastCharacter) := tmp;
  47     string += 1;
  48     pointerToLastCharacter -= 1;
  49   EndWhile
  50 EndFunction

  52 Function convertIntegerToString(PointerToCharacter string, Integer32 number) 
  53     Which Returns Nothing Does
  54   Integer32 isNumberNegative := 0;
  55   If number < 0 Then
  56     number *= -1;
  57     isNumberNegative := 1;
  58   EndIf
  59   Integer32 i := 0;
  60   While number >= 10 Loop
  61     ValueAt(string + i) := '0' + mod(number, 10);
  62     number /= 10;
  63     i += 1;
  64   EndWhile
  65   ValueAt(string + i) := '0' + number;
  66   i += 1;
  67   If isNumberNegative Then 
  68     ValueAt(string + i) := '-';
  69     i += 1;
  70   EndIf
  71   ValueAt(string + i) := 0;
  72   reverseString(string);
  73 EndFunction

  89 Function isNodeWithinBounds(
  90          PointerToNode node) Which Returns Integer32 Does
  91   Return AddressOf(nodes[0]) <= node <= AddressOf(nodes[128 - 1]) and
  92       mod(node - AddressOf(nodes[0]), SizeOf(Node)) = 0;
  93 EndFunction

  95 Function newNode() Which Returns PointerToNode Does
  96   Integer16 i := 0;
  97   While i < 128 Loop
  98     If not(isNodeUsed[i]) Then
  99       nodes[i].key := 0;
 100       nodes[i].left :=
 101          nodes[i].right :=
 102            PointerToNode(0);
 103       nodes[i].x := nodes[i].y := 0;
 104       isNodeUsed[i] := 1;
 105       Return AddressOf(nodes[i]);
 106     EndIf
 107     i += 1;
 108   EndWhile
 109   noMoreFreeMemory();
 110   asm("unreachable");
 111 EndFunction

 113 Function freeNode(
 114     PointerToNode node) Which Returns Nothing Does
 115   If not(isNodeWithinBounds(node)) Then
 116     segmentationFault();
 117   EndIf
 118   isNodeUsed[
 119     (node - AddressOf(nodes[0])) / SizeOf(Node)] := 0;
 120 EndFunction

 124 Function InsertBST(PointerToNode node, Integer32 key) 
 125     Which Returns PointerToNode Does
 126   If not(node) Then
 127     PointerToNode newNode := newNode();
 128     newNode->key := key;
 129     Return newNode;
 130   EndIf
 131   If node->key > key Then
 132     node->left := InsertBST(node->left, key);
 133   ElseIf node->key < key Then
 134     node->right := InsertBST(node->right, key);
 135   EndIf
 136   Return node;
 137 EndFunction

 174 Function Splay(PointerToNode node, Integer32 X)
 175     Which Returns PointerToNode Does
 176   If not(node) or node->key = X Then
 177     Return node;
 178   ElseIf node->key > X Then
 179     If not(node->left) Then
 180       Return node;
 181     ElseIf node->left->key > X Then
 182       node->left->left := Splay(node->left->left, X);
 183       node := RRotate(node);
 184     ElseIf node->left->key < X Then
 185       node->left->right := Splay(node->left->right, X);
 186       If node->left->right Then
 187         node->left := LRotate(node->left);
 188       EndIf
 189     EndIf
 190     If not(node->left) Then
 191       Return node;
 192     Else
 193       Return RRotate(node);
 194     EndIf
 195   Else
 196     If not(node->right) Then
 197       Return node;
 198     ElseIf node->right->key > X Then
 199       node->right->left := Splay(node->right->left, X);
 200       If node->right->left Then
 201         node->right := RRotate(node->right);
 202       EndIf
 203     ElseIf node->right->key < X Then
 204       node->right->right := Splay(node->right->right, X);
 205       node := LRotate(node);
 206     EndIf
 207     If not(node->right) Then
 208       Return node;
 209     Else
 210       Return LRotate(node);
 211     EndIf
 212   EndIf
 213   asm("unreachable");
 214 EndFunction

 216 Function Insert(PointerToNode node, Integer32 X)
 217     Which Returns PointerToNode Does
 218   If not(node) Then
 219     PointerToNode newnode := newNode();
 220     newnode->key := X;
 221     Return newnode;
 222   EndIf
 223   node := Splay(node, X);
 224   If node->key = X Then
 225     Return node;
 226   EndIf
 227   PointerToNode newnode := newNode();
 228   newnode->key := X;
 229 
 230   If node->key > X Then
 231     newnode->right := node;
 232     newnode->left := node->left;
 233     node->left := PointerToNode(0);
 234   Else
 235     newnode->left := node;
 236     newnode->right := node->right;
 237     node->right := PointerToNode(0);
 238   EndIf
 239   Return newnode;
 240 EndFunction

 242 Function Delete(PointerToNode node, Integer32 X)
 243     Which Returns PointerToNode Does
 244   PointerToNode temp;
 245   If not(node) Then
 246     Return node;
 247   EndIf
 248   node := Splay(node, X);
 249   If not(X = node->key) Then
 250     Return node;
 251   EndIf
 252   If not(node->left) Then
 253     temp := node;
 254     node := node->right;
 255   Else
 256     temp := node;
 257     node := Splay(node->left, X);
 258     node->right := temp->right;
 259   EndIf
 260   freeNode(temp);
 261   Return node;
 262 EndFunction

 296 Function assignCoordinatesInorder(PointerToNode node,
 297                                   Integer32 depth,
 298                                   Integer32 horizontalSpacing,
 299                                   Integer32 verticalSpacing)
 300     Which Returns Nothing Does
 301   If not(node) Then
 302     Return;
 303   EndIf
 304   assignCoordinatesInorder(node->left,
 305                            depth + 1,
 306                            horizontalSpacing,
 307                            verticalSpacing);
 308   node->x := inorderCounter * horizontalSpacing;
 309   node->y := depth * verticalSpacing;
 310   inorderCounter += 1;
 311   assignCoordinatesInorder(node->right,
 312                            depth + 1,
 313                            horizontalSpacing,
 314                            verticalSpacing);

 321 Function findMinMaxX(PointerToNode node,
 322                      PointerToMaxima m) 
 323     Which Returns Nothing Does
 324   If not(node) Then
 325     Return;

 383 Function render() Which Returns Nothing Does
 384   clearScreen();
 385   root := PointerToNode(0);
 386   Integer32 i := 0;
 387   While i < numberOfKeys Loop
 388     If insert_or_delete[i] = 'I' Then
 389       Character stringToBePrinted[32] := {0};
 390       strcat(AddressOf(stringToBePrinted[0]), "Inserting ");
 391       appendIntegerToString(
 392         AddressOf(stringToBePrinted[0]),
 393         keys[i]);
 394       strcat(AddressOf(stringToBePrinted[0]), "...\n");
 395       printString(AddressOf(stringToBePrinted[0]));
 396 
 397       root := Insert(root, keys[i]);
 398     ElseIf insert_or_delete[i] = 'D' Then
 399       Character stringToBePrinted[32] := {0};
 400       strcat(AddressOf(stringToBePrinted[0]), "Deleting ");
 401       appendIntegerToString(
 402         AddressOf(stringToBePrinted[0]),
 403         keys[i]);
 404       strcat(AddressOf(stringToBePrinted[0]), "...\n");
 405       printString(AddressOf(stringToBePrinted[0]));
 406 
 407       root := Delete(root, keys[i]);
 408     ElseIf insert_or_delete[i] = 'S' Then 
 409       Character stringToBePrinted[32] := {0};
 410       strcat(AddressOf(stringToBePrinted[0]), "Searching for ");
 411       appendIntegerToString(
 412         AddressOf(stringToBePrinted[0]),
 413         keys[i]);
 414       strcat(AddressOf(stringToBePrinted[0]), "...\n");
 415       printString(AddressOf(stringToBePrinted[0]));
 416 
 417       root := Search(root, keys[i]);
 418     ElseIf insert_or_delete[i] = 'B' Then
 419       Character stringToBePrinted[64] := {0};
 420       strcat(AddressOf(stringToBePrinted[0]), "Inserting ");
 421       appendIntegerToString(
 422         AddressOf(stringToBePrinted[0]),
 423         keys[i]);
 424       strcat(AddressOf(stringToBePrinted[0]), " as if we were doing a Binary Search Tree...\n");
 425       printString(AddressOf(stringToBePrinted[0]));
 426 
 427       root := InsertBST(root, keys[i]);
 428     Else
 429       printString(R"RawString(The character array "insert_or_delete" can only contain characters 'I', 'B', 'S', and 'D'!
 430 )RawString");
 431       asm("unreachable");
 432     EndIf
 433     i += 1;
 434   EndWhile
 435 
 436   inorderCounter := 1;
 437   assignCoordinatesInorder(root, 0, 110, 90);
 438 
 439   InstantiateStructure Maxima globalMax;
 440   globalMax.minimumX := globalMax.maximumX := 0;
 441   If root Then
 442     globalMax.minimumX := root->x;
 443     globalMax.maximumX := root->x;
 444     findMinMaxX(root, AddressOf(globalMax));
 445   EndIf
 446 
 447   Integer32 margin := 20;
 448   Integer32 diagramWidth := (globalMax.maximumX - globalMax.minimumX) + 120 + margin * 2;
 449   Integer32 diagramHeight := (nodeHeight(root) + 1) * 90 + margin;
 450 
 451   setDiagramWidth(diagramWidth);
 452   setDiagramHeight(diagramHeight);
 453 
 454   drawTree(root, globalMax.minimumX - margin);
 455 
 456   freeTree(root);
 457   printString("Splay Tree rendered.\n");
 458 EndFunction
