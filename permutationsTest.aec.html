  12 //JavaScript equivalent of C "strlen" is the "length" property of a string
  13 // and there is, as far as I know, no way to call it from outside of JS.
  14 //Nevertheless, I think I can easily write it myself.
  15 Function strlen(CharacterPointer ptr) Which Returns Integer32 Does
  16     Return ValueAt(ptr) = 0 ?
  17                                 0
  18                             :
  19                                 1 + strlen(ptr + 1);
  20 EndFunction

  22 Integer32 originalNumberOfDigits[10];
  23 Integer32 NDEBUG := 1;
  24 Integer32 numberOfPermutations;
  25 
  26 Function recursiveProcedure(CharacterPointer currentAttempt)
  27     Which Returns Nothing Does
  28     Integer32 lengthOfTheCurrentAttempt := strlen(currentAttempt);
  29     If not(NDEBUG) Then
  30         printString(
  31             "DEBUG: \"recursiveProcedure\" has been invoked with the argument: \""
  32             );
  33         printString(currentAttempt);
  34         printString("\". \"strlen\" says it has length of ");
  35         printInt(lengthOfTheCurrentAttempt);
  36         printString(".\n");
  37     EndIf
  38     Integer32 currentNumberOfDigits[10] :=
  39         {0, 0, 0, 0, 0,
  40          0, 0, 0, 0, 0};
  41     Integer32 i := 0;
  42     While i<lengthOfTheCurrentAttempt Loop
  43         currentNumberOfDigits[ValueAt(currentAttempt + i) - '0'] :=
  44             currentNumberOfDigits[ValueAt(currentAttempt + i) - '0'] + 1;
  45         i := i + 1;
  46     EndWhile
  47     If not(NDEBUG) Then
  48         i := 0;
  49         While i < 10 Loop
  50             printString("DEBUG: The current number has ");
  51             printInt(currentNumberOfDigits[i]);
  52             printString(" digits '");
  53             printInt(i);
  54             printString("'.\n");
  55             i := i + 1;
  56         EndWhile
  57     EndIf
  58     i := 0;
  59     While i < 10 Loop
  60         If currentNumberOfDigits[i] > originalNumberOfDigits[i] Then
  61             If not(NDEBUG) Then
  62                 printString("DEBUG: There are too many digits '");
  63                 printInt(i);
  64                 printString("', ending \"recursiveProcedure\".\n");
  65             EndIf
  66             Return;
  67         EndIf
  68         i := i + 1;
  69     EndWhile
  70     Integer32 haveWeFoundAPermutation := 1;
  71     i := 0;
  72     While i < 10 Loop
  73         If currentNumberOfDigits[i] < originalNumberOfDigits[i] Then
  74             haveWeFoundAPermutation := 0;
  75         EndIf
  76         i := i + 1;
  77     EndWhile
  78     If haveWeFoundAPermutation Then
  79         printString("Permutation #");
  80         numberOfPermutations:=numberOfPermutations+1;
  81         printInt(numberOfPermutations);
  82         printString(": ");
  83         printString(currentAttempt);
  84         printString("\n");
  85         Return;
  86     EndIf
  87     Character digitWeAreAdding := '0';
  88     While digitWeAreAdding < '9' + 1 Loop //The less-than-or-equal operator
  89                                           //">=" hasn't yet been implemented.
  90         Character newAttempt[12];
  91         i := 0;
  92         While i < 12 Loop
  93             If i < lengthOfTheCurrentAttempt Then
  94                 newAttempt[i] := ValueAt(currentAttempt + i);
  95             Else
  96                 newAttempt[i] := 0;
  97             EndIf
  98             i := i + 1;
  99         EndWhile
 100         newAttempt[lengthOfTheCurrentAttempt] := digitWeAreAdding;
 101         If currentNumberOfDigits[digitWeAreAdding - '0'] < 
 102             originalNumberOfDigits[digitWeAreAdding - '0'] Then //To speed
 103                                                                 //things up
 104                                                                 //a bit.
 105             recursiveProcedure(AddressOf(newAttempt[0]));
 106         EndIf
 107         digitWeAreAdding := digitWeAreAdding + 1;
 108     EndWhile
 109 EndFunction

 111 Function printPermutationsOfDigits(Integer32 original)
 112     Which Returns Nothing Does
 113     clearScreen();
 114     If original < 0 Then
 115         original := -original;
 116     EndIf
 117     printString("Printing all the permutations of digits of the number: ");
 118     printInt(original); //Unfortunately, the JavaScript standard library
 119                         //doesn't have "printf".
 120     printString("\n");
 121     Integer32 i := 0;
 122     While i < 10 Loop
 123         originalNumberOfDigits[i] :=     0;
 124         i                         := i + 1;
 125     EndWhile
 126     If original = 0 Then
 127         originalNumberOfDigits[0] := 1;
 128     EndIf
 129     While original > 0 Loop
 130         originalNumberOfDigits[mod(original, 10)] :=
 131             originalNumberOfDigits[mod(original, 10)] + 1;
 132         original := original / 10;
 133     EndWhile
 134     If not(NDEBUG) Then
 135         i := 0;
 136         While i < 10 Loop
 137             printString("DEBUG: The original number has ");
 138             printInt(originalNumberOfDigits[i]);
 139             printString(" digits '");
 140             printInt(i);
 141             printString("'.\n");
 142             i := i + 1;
 143         EndWhile
 144     EndIf
 145     numberOfPermutations := 0;
 146     recursiveProcedure("");
 147     printString("The end!");
 148 EndFunction
